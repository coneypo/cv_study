25  * @brief This class represents an object to work with different parameters
26  */
27 class Parameter {
28 public:
29  /**
30  * @brief Default constructor
31  */
32  Parameter() = default;
33 
34  /**
35  * @brief Move constructor
36  * @param parameter Parameter object
37  */
38  Parameter(Parameter &&parameter) noexcept {
39  std::swap(ptr, parameter.ptr);
40  }
41 
42  /**
43  * @brief Copy constructor
44  * @param parameter Parameter object
45  */
46  Parameter(const Parameter &parameter) {
47  *this = parameter;
48  }
49 
50  /**
51  * @brief Constructor creates parameter with object
52  * @tparam T Parameter type
53  * @tparam U Identity type-transformation
54  * @param parameter object
55  */
56  template<class T, typename = typename std::enable_if<!std::is_same<typename std::decay<T>::type, Parameter>::value>::type >
57  Parameter(T&& parameter) { // NOLINT
58  static_assert(!std::is_same<typename std::decay<T>::type, Parameter>::value, "To prevent recursion");
59  ptr = new RealData<typename std::decay<T>::type>(std::forward<T>(parameter));
60  }
61 
62  /**
63  * @brief Constructor creates string parameter from char *
64  * @param str char array
65  */
66  Parameter(const char *str): Parameter(std::string(str)) {} // NOLINT
67 
68  /**
69  * @brief Destructor
70  */
71  virtual ~Parameter() {
72  clear();
73  }
74 
75  /**
76  * Copy operator for Parameter
77  * @param parameter Parameter object
78  * @return Parameter
79  */
80  Parameter& operator=(const Parameter& parameter) {
81  if (this == &parameter) {
82  return *this;
83  }
84  clear();
85  if (!parameter.empty())
86  ptr = parameter.ptr->copy();
87  return *this;
88  }
89 
90  /**
91  * Remove a value from parameter
92  */
93  void clear() {
94  delete ptr;
95  ptr = nullptr;
96  }
97 
98  /**
99  * Checks that parameter contains a value
100  * @return false if parameter contains a value else false
101  */
102  bool empty() const noexcept {
103  return nullptr == ptr;
104  }
105 
106  /**
107  * Checks the type of value
108  * @tparam T Type of value
109  * @return true if type of value is correct
110  */
111  template<class T>
112  bool is() const {
113  return empty() ? false : ptr->is(typeid(T));
114  }

117  * Dynamic cast to specified type
118  * @tparam T type
119  * @return casted object
120  */
121  template<typename T>
122  T &&as() && {
123  return std::move(dyn_cast<T>(ptr));
124  }

131  template<class T>

183  /**
184  * @brief The comparison operator for the Parameter
185  * @param rhs object to compare
186  * @return true if objects are equal
187  */
188  bool operator == (const Parameter& rhs) const {
189  return *ptr == *(rhs.ptr);
190  }
191  /**
192  * @brief The comparison operator for the Parameter
193  * @param rhs object to compare
194  * @return true if objects aren't equal
195  */
196  bool operator != (const Parameter& rhs) const {
197  return !(*this == rhs);
198  }

200 private:
201  template<class T, class EqualTo>
202  struct CheckOperatorEqual {
203  template<class U, class V>
204  static auto test(U*) -> decltype(std::declval<U>() == std::declval<V>()) {
205  return false;
206  }
207 
208  template<typename, typename>
209  static auto test(...) -> std::false_type {
210  return {};
211  }
212 
213  using type = typename std::is_same<bool, decltype(test<T, EqualTo>(nullptr))>::type;
214  };
215 
216  template<class T, class EqualTo = T>
217  struct HasOperatorEqual : CheckOperatorEqual<T, EqualTo>::type {};
218 
219  struct Any {
220  virtual ~Any() = default;
221  virtual bool is(const std::type_info&) const = 0;
222  virtual Any *copy() const = 0;
223  virtual bool operator==(const Any& rhs) const = 0;
224  };
225 
226  template<class T>
227  struct RealData: Any, std::tuple<T> {
228  using std::tuple<T>::tuple;
229 
230  bool is(const std::type_info& id) const override {
231  return id == typeid(T);
232  }
233  Any *copy() const override {
234  return new RealData{get()};
235  }
236 
237  T& get() & {
238  return std::get<0>(*static_cast<std::tuple<T>*>(this));
239  }
240 
241  const T& get() const & {
242  return std::get<0>(*static_cast<const std::tuple<T>*>(this));
243  }
244 
245  template <class U>
246  typename std::enable_if<!HasOperatorEqual<U>::value, bool>::type
247  equal(const Any& left, const Any& rhs) const {
248  THROW_IE_EXCEPTION << "Parameter doesn't contain equal operator";
249  }
250 
251  template <class U>
252  typename std::enable_if<HasOperatorEqual<U>::value, bool>::type
253  equal(const Any& left, const Any& rhs) const {
254  return dyn_cast<U>(&left) == dyn_cast<U>(&rhs);
255  }
256 
257  bool operator==(const Any& rhs) const override {
258  return rhs.is(typeid(T)) && equal<T>(*this, rhs);
259  }
260  };
261 
262  template<typename T>
263  static T &dyn_cast(Any* obj) {
264  if (obj == nullptr)
265  THROW_IE_EXCEPTION << "Parameter is empty!";
266  return dynamic_cast<RealData<T>&>(*obj).get();
267  }
268 
269  template<typename T>
270  static const T &dyn_cast(const Any* obj) {
271  if (obj == nullptr)
272  THROW_IE_EXCEPTION << "Parameter is empty!";
273  return dynamic_cast<const RealData<T> &>(*obj).get();
274  }
275 
276  Any *ptr = nullptr;
277 };
