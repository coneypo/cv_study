35 /**
36  * @brief Gets the top n results from a tblob
37  * @param n Top n count
38  * @param input 1D tblob that contains probabilities
39  * @param output Vector of indexes for the top n places
40  */
41 template<class T>
42 inline void TopResults(unsigned int n, TBlob<T> &input, std::vector<unsigned> &output) {
43  SizeVector dims = input.getTensorDesc().getDims();
44  size_t input_rank = dims.size();
45  if (!input_rank || !dims[0])
46  THROW_IE_EXCEPTION << "Input blob has incorrect dimensions!";
47  size_t batchSize = dims[0];
48  std::vector<unsigned> indexes(input.size() / batchSize);
49 
50  n = static_cast<unsigned>(std::min<size_t>((size_t) n, input.size()));
51 
52  output.resize(n * batchSize);
53 
54  for (size_t i = 0; i < batchSize; i++) {
55  size_t offset = i * (input.size() / batchSize);
56  T *batchData = input.data();
57  batchData += offset;
58 
59  std::iota(std::begin(indexes), std::end(indexes), 0);
60  std::partial_sort(std::begin(indexes), std::begin(indexes) + n, std::end(indexes),
61  [&batchData](unsigned l, unsigned r) {
62  return batchData[l] > batchData[r];
63  });
64  for (unsigned j = 0; j < n; j++) {
65  output.at(i * n + j) = indexes.at(j);
66  }
67  }
68 }

70 #define TBLOB_TOP_RESULT(precision)\
71  case InferenceEngine::Precision::precision : {\
72  using myBlobType = InferenceEngine::PrecisionTrait<Precision::precision>::value_type;\
73  TBlob<myBlobType> &tblob = dynamic_cast<TBlob<myBlobType> &>(input);\
74  TopResults(n, tblob, output);\
75  break;\
76  }

101 /**
102  * @brief Copies a 8-bit RGB image to the blob.
103  * Throws an exception in case of dimensions or input size mismatch
104  * @tparam data_t Type of the target blob
105  * @param RGB8 8-bit RGB image
106  * @param RGB8_size Size of the image
107  * @param blob Target blob to write image to
108  */
109 template<typename data_t>
110 void copyFromRGB8(uint8_t *RGB8, size_t RGB8_size, InferenceEngine::TBlob<data_t> *blob) {
111  SizeVector dims = blob->getTensorDesc().getDims();
112  if (4 != dims.size())
113  THROW_IE_EXCEPTION << "Cannot write data to input blob! Blob has incorrect dimensions size "
114  << dims.size();
115  size_t num_channels = dims[1]; // because RGB
116  size_t num_images = dims[0];
117  size_t w = dims[3];
118  size_t h = dims[2];
119  size_t nPixels = w * h;
120 
121  if (RGB8_size != w * h * num_channels * num_images)
122  THROW_IE_EXCEPTION << "input pixels mismatch, expecting " << w * h * num_channels * num_images
123  << " bytes, got: " << RGB8_size;
124 
125  std::vector<data_t *> dataArray;
126  for (unsigned int n = 0; n < num_images; n++) {
127  for (unsigned int i = 0; i < num_channels; i++) {
128  if (!n && !i && dataArray.empty()) {
129  dataArray.push_back(blob->data());
130  } else {
131  dataArray.push_back(dataArray.at(n * num_channels + i - 1) + nPixels);
132  }
133  }
134  }
135  for (size_t n = 0; n < num_images; n++) {
136  size_t n_num_channels = n * num_channels;
137  size_t n_num_channels_nPixels = n_num_channels * nPixels;
138  for (size_t i = 0; i < nPixels; i++) {
139  size_t i_num_channels = i * num_channels + n_num_channels_nPixels;
140  for (size_t j = 0; j < num_channels; j++) {
141  dataArray.at(n_num_channels + j)[i] = RGB8[i_num_channels + j];
142  }
143  }
144  }
145 }

147 /**
148  * @brief Splits the RGB channels to either I16 Blob or float blob.
149  * The image buffer is assumed to be packed with no support for strides.
150  * @param imgBufRGB8 Packed 24bit RGB image (3 bytes per pixel: R-G-B)
151  * @param lengthbytesSize Size in bytes of the RGB image. It is equal to amount of pixels times 3 (number of channels)
152  * @param input Blob to contain the split image (to 3 channels)
153  */
154 inline void ConvertImageToInput(unsigned char *imgBufRGB8, size_t lengthbytesSize, Blob &input) {
155  TBlob<float> *float_input = dynamic_cast<TBlob<float> *>(&input);
156  if (float_input != nullptr) copyFromRGB8(imgBufRGB8, lengthbytesSize, float_input);
157 
158  TBlob<short> *short_input = dynamic_cast<TBlob<short> *>(&input);
159  if (short_input != nullptr) copyFromRGB8(imgBufRGB8, lengthbytesSize, short_input);
160 
161  TBlob<uint8_t> *byte_input = dynamic_cast<TBlob<uint8_t> *>(&input);
162  if (byte_input != nullptr) copyFromRGB8(imgBufRGB8, lengthbytesSize, byte_input);
163 }

165 /**
166  * @brief Copies data from a certain precision to float
167  * @param dst Pointer to an output float buffer, must be allocated before the call
168  * @param src Source blob to take data from
169  */
170 template<typename T>
171 void copyToFloat(float *dst, const InferenceEngine::Blob *src) {
172  if (!dst) {
173  return;
174  }
175  const InferenceEngine::TBlob<T> *t_blob = dynamic_cast<const InferenceEngine::TBlob<T> *>(src);
176  if (t_blob == nullptr) {
177  THROW_IE_EXCEPTION << "input type is " << src->getTensorDesc().getPrecision() << " but input is not " << typeid(T).name();
178  }
179 
180  const T *srcPtr = t_blob->readOnly();
181  if (srcPtr == nullptr) {
182  THROW_IE_EXCEPTION << "Input data was not allocated.";
183  }
184  for (size_t i = 0; i < t_blob->size(); i++) dst[i] = srcPtr[i];
185 }
